calculating a critical value

qt(p = .05/2, df = 40)

calculating t-score

# population mean
mu0 <- 2

# number of observations
n <- 41

# sample mean
xbar <- mean(acorns)

# sample standard deviation
s <- sd(acorns)

# sample standard error
se <- s/sqrt(n)

# degrees of freedom
df <- n - 1

# t-score
t <- (xbar-mu0)/se

t
[1] 1.803471
