calculating t-score

Code 
# population mean
mu0 <- 2

# number of observations
n <- 20

# sample mean
ybar <- mean(leaves)

# sample standard deviation
s <- sd(leaves)

# sample standard error
se <- s/sqrt(n)

# degrees of freedom
df <- n - 1

# t-score
t <- (ybar-mu0)/se

t

\[ SE = \frac{s}{\sqrt{n}} = \frac{0.371}{4.47} = 0.08 \]

nA <- length(x)
nB <- length(y)

(xa - xb)/sqrt((vara/nA)+(varb/nB))
(nA - 1) + (nB - 1)
